// index is position of character in string, the count starts from 0, you can check position of character using its index in brackets [1]

let greeting = "hello";

console.log(greeting[1]);

// "e"

// to check the last character you can do String.length - 1 (because index count starts with 0)

let greeting = "hello";

console.log(greeting[greeting.length - 1]);

// "o"

let poem = "Roses are red,\nViolets are blue.";

// \n is used for a newline

console.log(poem);

/* Roses are red,

                             Violets are blue.*/

let statement = "She said, \"Hello!\"";

// place \ before special characters (", \, \\) for them to display

console.log(statement);

// She said, "Hello!"

const name = "Alice";

const age = 25

const deathAge = 79

// average life expectancy in US

const greeting = `${name}, ${age}! you have ${deathAge - age} years until to death!`;

// actually + operator sucks, use ` ` . and the best part? \n is not needed!

console.log(greeting);

// Alice, 25! you have 54 years until to death!
